        messages: [
          {role: 'system', content: '你是OpenAI开发的AI助手'},
          {role: 'user', content: '请解释OpenAI API中转站的优势'}
        ],
        temperature: 0.7
      },
      {
        headers: {
          'Authorization': `Bearer ${API_KEY}`,
          'Content-Type': 'application/json'
        }

  } catch (error) {

  }
}

callOpenAI();
